persona(dni PRIMARY KEY,nacionalidad,nombre,apellido)
tipo_delito(id PRIMARY KEY,nombre,descripcion)

antecedente(id PRIMARY KEY,descripcion,fecha,tipo_delito_id,dni,FOREIGN KEY(tipo_delito_id) REFERENCES tipo_delito(id),FOREIGN KEY(dni) REFERENCES persona(dni))

persona_con_licencia(dni PRIMARY KEY,		fecha_vencimiento)

cedula_vehicular(codigo PRIMARY KEY,tipo_cedula ENUM('Verde','Azul'))

tipo_vehiculo(id PRIMARY KEY,nombre)

categoria_vehiculo(id PRIMARY KEY,nombre)

vehiculo(matricula PRIMARY KEY,modelo SMALLINT,detalle,	tipo_id,categoria_id,dni,FOREIGN KEY(dni) REFERENCES persona(dni),FOREIGN KEY(tipo_id) REFERENCES tipo_vehiculo(id),FOREIGN KEY(categoria_id) REFERENCES categoria_vehiculo(id)	)

registro_automotor(dni,matricula,cedula_codigo,PRIMARY KEY(dni,cedula_codigo),FOREIGN KEY(dni) REFERENCES persona(dni),FOREIGN KEY(matricula) REFERENCES vehiculo(matricula),FOREIGN KEY(cedula_codigo) REFERENCES cedula_vehicular(codigo))

tipo_cobertura(id PRIMARY KEY,nombre,descripcion)

compania_seguro(id PRIMARY KEY,nombre,descripcion)

cobertura(vehiculo_matricula,tipo_id,compania_id,PRIMARY KEY (vehiculo_matricula,tipo_id,compania_id),FOREIGN KEY(tipo_id) REFERENCES tipo_cobertura(id),FOREIGN KEY(compania_id) REFERENCES compania_seguro(id))
camino(id PRIMARY KEY,nombre,longitud)

direccion(	altura,camino_id,PRIMARY KEY(altura,camino_id),FOREIGN KEY(camino_id) REFERENCES camino(id))

infraccion(id PRIMARY KEY,descripcion,fecha,vehiculo_matricula,dni,	direccion_altura,direccion_camino_id,FOREIGN KEY(dni) REFERENCES persona(dni),FOREIGN KEY(direccion_altura,direccion_camino_id) REFERENCES direccion(altura,camino_id),FOREIGN KEY(vehiculo_matricula) REFERENCES vehiculo(matricula))

tipo_infraccion(id PRIMARY KEY,nombre,descripcion)

tipo_accidente(id PRIMARY KEY, nombre, descripcion)

tipo_falla_humana (id PRIMARY KEY,nombre, descripcion)

comisaria(numero PRIMARY KEY,direccion_altura,direccion_camino_id,FOREIGN KEY(direccion_altura,direccion_camino_id) REFERENCES direccion(altura,camino_id))


denuncia(id PRIMARY KEY,fecha,comisaria_nro,FOREIGN KEY(comisaria_nro) REFERENCES comisaria(numero)) 

colision(id PRIMARY KEY,descripcion)

tipo_peritaje(id PRIMARY KEY,nombre,descripcion)


peritaje(id PRIMARY KEY,fecha,descripcion,conclusion,tipo_peritaje_id,FOREIGN KEY(tipo_peritaje_id) REFERENCES tipo_peritaje(id))

siniestro(id PRIMARY KEY,descripcion,fecha_hora,	tipo_falla_id,tipo_accidente_id, denuncia_id,colision_id,direccion_altura,direccion_camino_id,FOREIGN KEY(tipo_accidente_id) REFERENCES tipo_accidente(id),FOREIGN KEY(tipo_falla_id) REFERENCES tipo_falla_humana(id),FOREIGN KEY(denuncia_id) REFERENCES denuncia(id),FOREIGN KEY(colision_id) REFERENCES colision(id),FOREIGN KEY(direccion_altura,direccion_camino_id) REFERENCES direccion(altura,camino_id))

conclusion(id PRIMARY KEY,conclusion_final,siniestro_id,FOREIGN KEY(siniestro_id) REFERENCES siniestro(id))

victima(conclusion_id,dni,PRIMARY KEY(conclusion_id,dni),FOREIGN KEY(conclusion_id) REFERENCES conclusion(id),FOREIGN KEY(dni) REFERENCES persona(dni))

victimario(conclusion_id,dni,PRIMARY KEY(conclusion_id,dni),FOREIGN KEY(conclusion_id) REFERENCES conclusion(id),FOREIGN KEY(dni) REFERENCES persona(dni))

testigos(siniestro_id,dni,FOREIGN KEY(dni) REFERENCES persona(dni),FOREIGN KEY(siniestro_id) REFERENCES siniestro(id))

provincia(id PRIMARY KEY,nombre)

localidad(id PRIMARY KEY,nombre,provincia_id,FOREIGN KEY(provincia_id) REFERENCES provincia(id))

registro_calles(camino_id,localidad_id,altura_desde,altura_hasta,PRIMARY KEY(camino_id,localidad_id),FOREIGN KEY(camino_id) REFERENCES camino(id),FOREIGN KEY(localidad_id) REFERENCES localidad(id))

tipo_camino(id PRIMARY KEY,nombre)

clasificacion_camino(camino_id,tipo_id,altura_hasta,altura_desde,longitud,PRIMARY KEY(camino_id,tipo_id,altura_desde,altura_hasta),FOREIGN KEY(camino_id) REFERENCES camino(id),FOREIGN KEY(tipo_id) REFERENCES tipo_camino(id))
